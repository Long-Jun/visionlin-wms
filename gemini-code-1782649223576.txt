【系統角色與任務目標】
你現在是一位精通 Google Firebase (Firestore) 模組化 SDK (v10) 與純前端開發的高級全端工程師。
請幫我開發／維護一個「單一 HTML 檔案」的線上版倉庫 ERP 系統。不需要後端伺服器，資料庫完全依賴 Firebase Cloud Firestore，並且必須內建嚴格的「交易鎖定（Transaction）」來防止多人同時操作時的庫存打架（Race Condition）。

【專案背景與環境】
* 技術棧：純 HTML、CSS、Vanilla JavaScript (ES6 module)。
* Firebase 版本：v10 模組化寫法（initializeApp, getFirestore, runTransaction 等）。
* 實體倉庫配置：分為 B1 與 1F 兩層樓。層架以 A-N 命名，層數介於 3~5 層。儲位命名規則為 樓層-層架-層數（例如：B1-A-01）。

【Firestore 資料庫結構 (NoSQL Schema)】
1. locations (儲位主檔)
   - ID: 儲位代碼
   - Fields: id, floor, shelf, layer, status
2. items (物料主檔)
   - ID: 料號
   - Fields: id, name, spec, category, unit, safeStock, isControlled, returnable
3. inventory (庫存明細)
   - ID: itemId_locationId_status
   - Fields: itemId, locationId, quantity, status, lastUpdated
4. transactions (異動日誌)
   - ID: 隨機生成的交易序號
   - Fields: id, date, type, itemId, quantity, locationId, newLocationId, status, user, note

【核心商業邏輯要求 (極度重要)】
1. 防衝突出入庫 (Firestore runTransaction)：絕對不可以使用先 getDoc 再 setDoc 的普通寫法。必須使用 runTransaction 確保原子性（Atomicity）。
2. 批次操作 (WriteBatch)：初始化與結算盤點差異時，必須使用 writeBatch 進行整批寫入。
3. 盤點邏輯：結算時，只需針對「有差異」的品項執行批次修正，並生成異動紀錄。

【前端 UI/UX 規範】
1. 響應式（RWD），適合平板與 PC。
2. 單頁式設計（SPA），不重新載入網頁。
3. 對 Firebase 請求必須顯示 Loading 遮罩。

        async function processTransaction(e) {
            e.preventDefault();
            const type = document.getElementById('tx-type').value; const date = document.getElementById('tx-date').value.replace('T', ' ');
            const itemId = document.getElementById('tx-item').value.trim(); const qty = Number(document.getElementById('tx-qty').value);
            const loc = document.getElementById('tx-loc').value.trim(); const newLoc = document.getElementById('tx-new-loc').value.trim();
            const status = document.getElementById('tx-status').value; const user = document.getElementById('tx-user').value; const note = document.getElementById('tx-note').value;

            if(!cache.items.some(i => i.id === itemId)) return alert(`❌ 料號 ${itemId} 不存在！`);
            if(type !== 'out' && type !== 'borrow' && type !== 'return' && !cache.locations.some(l => l.id === loc)) return alert(`❌ 目標儲位 ${loc} 尚未啟用！`);
            if(type === 'move' && !cache.locations.some(l => l.id === newLoc)) return alert(`❌ 移位新目標儲位 ${newLoc} 尚未啟用！`);

            showLoading(true);
            try {
                await runTransaction(db, async (transaction) => {
                    const srcInvId = `${itemId}_${loc}_${status}`; const srcInvRef = doc(db, "inventory", srcInvId);
                    const srcSnap = await transaction.get(srcInvRef);
                    let currentSrcQty = srcSnap.exists() ? srcSnap.data().quantity : 0;

                    if(['out', 'move', 'borrow'].includes(type)) { if(!srcSnap.exists() || currentSrcQty < qty) throw new Error(`扣除餘量不足：${loc} 僅餘 ${currentSrcQty}。`); }

                    if(type === 'in') { transaction.set(srcInvRef, { itemId, locationId: loc, quantity: currentSrcQty + qty, status, lastUpdated: date }); }
                    else if(type === 'out') { const nextQty = currentSrcQty - qty; if(nextQty === 0) transaction.delete(srcInvRef); else transaction.update(srcInvRef, { quantity: nextQty, lastUpdated: date }); }
                    else if(type === 'move') {
                        const nextSrcQty = currentSrcQty - qty; if(nextSrcQty === 0) transaction.delete(srcInvRef); else transaction.update(srcInvRef, { quantity: nextSrcQty, lastUpdated: date });
                        const destInvId = `${itemId}_${newLoc}_${status}`; const destInvRef = doc(db, "inventory", destInvId);
                        const destSnap = await transaction.get(destInvRef); let currentDestQty = destSnap.exists() ? destSnap.data().quantity : 0;
                        transaction.set(destInvRef, { itemId, locationId: newLoc, quantity: currentDestQty + qty, status, lastUpdated: date });
                    }
                    else if(type === 'borrow') {
                        const nextSrcQty = currentSrcQty - qty; if(nextSrcQty === 0) transaction.delete(srcInvRef); else transaction.update(srcInvRef, { quantity: nextSrcQty, lastUpdated: date });
                        const borrowInvId = `${itemId}_${loc}_借出`; const borrowInvRef = doc(db, "inventory", borrowInvId);
                        const borrowSnap = await transaction.get(borrowInvRef); let currentBorrowQty = borrowSnap.exists() ? borrowSnap.data().quantity : 0;
                        transaction.set(borrowInvRef, { itemId, locationId: loc, quantity: currentBorrowQty + qty, status: '借出', lastUpdated: date });
                    }
                    else if(type === 'return') {
                        const borrowInvId = `${itemId}_${loc}_借出`; const borrowInvRef = doc(db, "inventory", borrowInvId);
                        const borrowSnap = await transaction.get(borrowInvRef); let currentBorrowQty = borrowSnap.exists() ? borrowSnap.data().quantity : 0;
                        if(!borrowSnap.exists() || currentBorrowQty < qty) throw new Error(`借出紀錄僅存 ${currentBorrowQty}，無法歸還 ${qty}。`);
                        if(currentBorrowQty - qty === 0) transaction.delete(borrowInvRef); else transaction.update(borrowInvRef, { quantity: currentBorrowQty - qty, lastUpdated: date });
                        transaction.set(srcInvRef, { itemId, locationId: loc, quantity: currentSrcQty + qty, status, lastUpdated: date });
                    }
                    const logId = `TX-${Date.now()}-${Math.random().toString(36).substr(2, 4)}`;
                    transaction.set(doc(db, "transactions", logId), { id: logId, date, type, itemId, quantity: qty, locationId: loc, newLocationId: type==='move'?newLoc:'', status: type==='borrow'?'借出':status, user, note });
                });
                alert("🟢 雲端交易成功！"); document.getElementById('tx-form').reset(); document.getElementById('tx-date').value = new Date().toISOString().slice(0,16); document.getElementById('tx-name').value = ''; await syncCache();
            } catch (err) { alert(`🔴 雲端作業失敗:\n${err.message}`); } finally { showLoading(false); }
        }

            if(!confirm(`確認執行雲端修復？`)) return; showLoading(true);

                const ref = doc(db, "inventory", `${c.itemId}_${c.locationId}_${c.status}`); const diff = c.realQty - c.quantity;
                if(c.realQty === 0) batch.delete(ref); else batch.set(ref, { itemId: c.itemId, locationId: c.locationId, quantity: c.realQty, status: c.status, lastUpdated: date });

        function triggerImportJson() {
            const fileInput = document.getElementById('import-file'); if(!fileInput.files.length) return alert("請選取備份檔案。");
            if(prompt("⚠️ 致命警告：請輸入大寫 'OVERWRITE' 以啟動程序：") !== 'OVERWRITE') return;
            showLoading(true); const reader = new FileReader();
            reader.onload = async function(e) {
                try {
                    const parsed = JSON.parse(e.target.result); if(!parsed.items || !parsed.locations || !parsed.inventory) throw new Error("結構缺失");
                    for (let col of ['items', 'locations', 'inventory', 'transactions']) { const snap = await getDocs(collection(db, col)); for (let docFile of snap.docs) { await deleteDoc(doc(db, col, docFile.id)); } }
                    for(let l of parsed.locations) { await setDoc(doc(db, "locations", l.id), l); }
                    for(let i of parsed.items) { await setDoc(doc(db, "items", i.id), i); }
                    for(let inv of parsed.inventory) { await setDoc(doc(db, "inventory", `${inv.itemId}_${inv.locationId}_${inv.status}`), inv); }
                    for(let tx of (parsed.transactions || [])) { await setDoc(doc(db, "transactions", tx.id), tx); }
                    alert("還原成功！系統即將刷新。"); location.reload();
                } catch(err) { alert("還原終止。"); showLoading(false); }
            }; reader.readAsText(fileInput.files[0]);
        }
